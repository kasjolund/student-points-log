    <script>

        const storedData = localStorage.getItem("students");

        let students = storedData ? JSON.parse(storedData) : [

            { name: "Student 1", points: 0 },

            { name: "Student 2", points: 0 },

            { name: "Student 3", points: 0 },

            { name: "Student 4", points: 0 },

            { name: "Student 5", points: 0 },

            { name: "Student 6", points: 0 },

            { name: "Student 7", points: 0 },

            { name: "Student 8", points: 0 }

        ];

        const pointValues = [2, 5, 10];

        const rewards = [

            { name: "Reward 1", cost: 50 },

            { name: "Reward 2", cost: 100 },

            { name: "Reward 3", cost: 150 }

        ];


        function saveData() {

            localStorage.setItem("students", JSON.stringify(students));

        }


        function renderStudents() {

            const container = document.getElementById("students");

            container.innerHTML = "";

            students.forEach((student, index) => {

                const studentDiv = document.createElement("div");

                studentDiv.classList.add("student");

                studentDiv.innerHTML = `

                    <input type="text" value="${student.name}" onchange="updateName(${index}, this.value)" />

                    <span class="points">${student.points} pts</span>

                    <div>

                        ${pointValues.map(value => `<button class="btn add" onclick="addPoints(${index}, ${value})">+${value}</button>`).join('')}

                    </div>

                    <div>

                        ${rewards.map(reward => `<button class="btn withdraw" onclick="withdrawPoints(${index}, ${reward.cost})">${reward.name} (-${reward.cost})</button>`).join('')}

                    </div>

                `;

                container.appendChild(studentDiv);

            });

        }


        function addPoints(index, points) {

            students[index].points += points;

            renderStudents();

            saveData();

        }


        function withdrawPoints(index, cost) {

            if (students[index].points >= cost) {

                if (confirm("Are you sure you want to redeem this reward?")) {

                    students[index].points -= cost;

                    renderStudents();

                    saveData();

                }

            } else {

                alert("Not enough points!");

            }

        }


        function updateName(index, newName) {

            students[index].name = newName;

            saveData();

        }


        renderStudents();
